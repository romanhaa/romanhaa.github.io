Distribution plot 1

Category: distribution

library(tidyverse)
library(wesanderson)

data(iris)

p <- ggplot(iris) +
  geom_jitter(aes(Species, Sepal.Width, color = Species), alpha = 0.75, size = 3, show.legend = FALSE) +
  geom_violin(aes(Species, Sepal.Width, fill = Species), alpha = 0.25, show.legend = FALSE) +
  geom_boxplot(aes(Species, Sepal.Width), outlier.size = 0, width = 0.2, alpha = 0.75, show.legend = FALSE) +
  scale_color_manual(values = wes_palette('BottleRocket2')) +
  scale_fill_manual(values = wes_palette('BottleRocket2')) +
  scale_y_continuous(labels = scales::comma) +
  theme_bw() +
  labs(y = 'Sepal width') +
  theme(axis.title.x = element_blank())
ggsave('1.png', p, height = 5, width = 6)
