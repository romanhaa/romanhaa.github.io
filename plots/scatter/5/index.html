# calculate density of dots as described on:
# https://slowkow.com/notes/ggplot2-color-by-density/
get_density <- function(x, y, ...) {
  dens <- MASS::kde2d(x, y, ...)
  ix <- findInterval(x, dens$x)
  iy <- findInterval(y, dens$y)
  ii <- cbind(ix, iy)
  return(dens$z[ii])
}

cells <- cells %>%
  mutate(density = get_density(nCount, nFeature, n = 100)) %>%
  arrange(density)

cells %>% glimpse()

p <- ggplot(cells, aes(nCount, nFeature)) +
  geom_point(aes(color = density), show.legend = FALSE) +
  scale_x_continuous(name = 'Number of transcripts', labels = scales::comma) +
  scale_y_continuous(name = 'Numner of expressed genes', labels = scales::comma) +
  scale_color_viridis() +
  theme_bw()
p <- ggMarginal(p, type = 'histogram', fill = 'grey', bins = 50)
